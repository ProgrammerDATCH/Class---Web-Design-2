9. Write a function that takes an array of 20 numbers and returns the median value


function findMedian() {
    const numbers = Array.from({length: 20}, (_, i) => i + 1);
    const sorted = numbers.sort((a, b) => a - b);
    const mid = Math.floor(sorted.length / 2);
    return sorted.length % 2 === 0 
        ? (sorted[mid - 1] + sorted[mid]) / 2
        : sorted[mid];
}

33. Write a function to convert a string to uppercase without using built-in methods


function customToUpper(str) {
    let result = '';
    for (let i = 0; i < str.length; i++) {
        let charCode = str.charCodeAt(i);
        if (charCode >= 97 && charCode <= 122) {
            result += String.fromCharCode(charCode - 32);
        } else {
            result += str[i];
        }
    }
    return result;
}

49. Write a JavaScript function to create a 3x3 table with numbers


function createTable() {
    const table = document.createElement('table');
    let counter = 1;
    
    for (let i = 0; i < 3; i++) {
        const row = table.insertRow();
        for (let j = 0; j < 3; j++) {
            const cell = row.insertCell();
            cell.textContent = counter++;
        }
    }
    
    const container = document.getElementById('table-container');
    container.innerHTML = '';
    container.appendChild(table);
}

50. Write a JavaScript function to validate a form


function validateForm() {
    const name = document.getElementById('form-name').value;
    const email = document.getElementById('form-email').value;
    const password = document.getElementById('form-password').value;
    
    if (!name || !email || !password) {
        alert('All fields must be filled out');
        return false;
    }
    
    if (!email.includes('@')) {
        alert('Please enter a valid email');
        return false;
    }
    
    alert('Form validated successfully!');
    return true;
}
